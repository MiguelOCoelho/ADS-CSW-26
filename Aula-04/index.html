      import math

      print("=== Cálculo de equação do segundo grau ===")
      print("Forma: ax² + bx + c = 0")

      # Entrada de dados
      a = float(input("Digite o valor de a: "))
      if a == 0:
          print("Erro: 'a' não pode ser zero (não é equação do segundo grau).")
      else:
          b = float(input("Digite o valor de b: "))
          c = float(input("Digite o valor de c: "))

          # Cálculo do delta
          delta = b**2 - 4*a*c

          if delta < 0:
              print("\nDelta < 0 → não há raízes reais.")
              print(f"Delta = {delta:.2f}")
          elif delta == 0:
              x = -b / (2*a)
              print("\nDelta = 0 → raízes reais iguais.")
              print(f"Delta = {delta:.2f}")
              print(f"x1 = x2 = {x:.2f}")
          else:
              x1 = (-b + math.sqrt(delta)) / (2*a)
              x2 = (-b - math.sqrt(delta)) / (2*a)
              print("\nDelta > 0 → duas raízes reais distintas.")
              print(f"Delta = {delta:.2f}")
              print(f"x1 = {x1:.2f}")
              print(f"x2 = {x2:.2f}")
